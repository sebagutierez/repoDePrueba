                }
            [/style]
        [/head]
        [/html]
        
  3. CSS EN UN DOCUMENTO CSS EXTERNO
            
            index.html
            [link rel="stylesheet" href="estilos.css"]
            
            estilos.css
            p{
                text-align:center;
                color: red;
            }

SELECTORES CSS

  • Nivel 5: {HTML: [p]} {CSS: p}
  • Nivel 4: {HTML: [div][p]} {CSS: div p}
  • Nivel 3: {HTML: [p class="uno"]} {CSS: .uno}
  • Nivel 2: {HTML: [p id="dos"]} {CSS: #dos}
  • Nivel 1: {HTML: [div id="tres"][p]} {CSS: #tres p}

DIMENSIONES EN CSS

  • {height: 10px;} Propiedad alto.
  • {width: 10px;} Propiedad ancho.
  • {min- height: 10px;} Propiedad mínima altura para que no desborde.
  • {max-width: 20px;} Propiedad máximo ancho.
  • {line-height: 20px;} Interlineado.
  • {font-size: 30px;} Tamaño del texto. 16px = 1rem = 1em = 100%
  • {color: black;} Color del texto.
  • {text-align: center;} Alineación del texto.
  • {background-color: black;} Color de fondo
  • {border: 2px solid black;} Linea de borde a un background.
